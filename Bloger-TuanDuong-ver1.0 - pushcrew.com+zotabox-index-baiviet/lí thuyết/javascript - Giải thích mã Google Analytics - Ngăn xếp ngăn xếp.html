Trước hết, tôi sẽ vượt qua điều này thông qua một người làm đẹp, ví dụ: http://jsbeautifier.org/

 (function (i, s, o, g, r, a, m) {
     i['GoogleAnalyticsObject'] = r;
     i[r] = i[r] || function () {
         (i[r].q = i[r].q || []).push(arguments)
     }, i[r].l = 1 * new Date();
     a = s.createElement(o),
     m = s.getElementsByTagName(o)[0];
     a.async = 1;
     a.src = g;
     m.parentNode.insertBefore(a, m)
 })(window, document, 'script', '//www.google-analytics.com/analytics.js', 'ga');

 ga('create', 'UA-xxxxxxxx-x', 'xxxxxx.com');
 ga('send', 'pageview');

Sau đó cho phép đánh giá việc đóng cửa 

(function (i, s, o, g, r, a, m) {
...
 })(window, document, 'script', '//www.google-analytics.com/analytics.js', 'ga');

bằng cách thay thế từng tham số được đặt tên: i, s, o, g, rvới các giá trị tương ứng của chúngwindow, document, 'script', '//www.google-analytics.com/analytics.js', 'ga'

Lưu ý rằng avà mcác thông số không có giá trị đầu vào và có nhiều như các biến kết quả.

Đây sẽ là khoảng (không làm phiền về phạm vi biến, vv) tương đương với

(function (i, s, o, g, r, a, m) {
     window['GoogleAnalyticsObject'] = 'ga';
     window['ga'] = window['ga'] || function () {
         (window['ga'].q = window['ga'].q || []).push(arguments)
     }, window['ga'].l = 1 * new Date();
     a = document.createElement('script'),
     m = document.getElementsByTagName('script')[0];
     a.async = 1;
     a.src = '//www.google-analytics.com/analytics.js';
     m.parentNode.insertBefore(a, m)
 })(window, document, 'script', '//www.google-analytics.com/analytics.js', 'ga');

 ga('create', 'UA-xxxxxxxx-x', 'xxxxxx.com');
 ga('send', 'pageview');

Tóm lại, mã này thực chất là gì, là nó tạo ra một thẻ script mới với dòng:

a = document.createElement('script'),

Sau đó, tìm thẻ tập lệnh đầu tiên

m = document.getElementsByTagName('script')[0];

Sau đó, nó đặt thẻ script mới được tạo ra để thực thi không đồng bộ (Có thể có thêm thông tin chi tiết về thực thi async tại Hiểu mã không đồng bộ trong các điều khoản của Layman nếu bạn cần)

a.async = 1;

1 trong dòng trên tương đương với true, nó được sử dụng 1 chỉ vì nó ngắn hơn.

Sau đó src của thẻ script này được thiết lập

 a.src = '//www.google-analytics.com/analytics.js';

Lưu ý rằng không có giao thức nào (http hoặc https) được chỉ định trong URL. Điều này sẽ cho phép tập lệnh được tải trong giao thức trình duyệt hiện tại.

Và cuối cùng nó được chèn vào trước thẻ script đầu tiên, do đó trình duyệt có thể bắt đầu tải nó.

 m.parentNode.insertBefore(a, m)

Vì vậy, để tóm tắt :

  1. Chúng tôi tạo thẻ tập lệnh
  2. Chúng tôi đặt nó để tải không đồng bộ async=true
  3. Chúng tôi chèn thẻ tập lệnh này, trước thẻ tập lệnh đầu tiên trong tài liệu

Các chi tiết cụ thể liên quan đến google analytics.

 window['ga'] = window['ga'] || function () {
     (window['ga'].q = window['ga'].q || []).push(arguments)
 }, window['ga'].l = 1 * new Date();

định nghĩa hàm toàn cầu có tên galà đẩy đối số của nó trong một hàng đợi Mảng (có tên q)

Sau đó, với các dòng

 ga('create', 'UA-xxxxxxxx-x', 'xxxxxx.com');
 ga('send', 'pageview');

nó đẩy các "sự kiện" này vào hàng đợi Array.

Khi kịch bản được tải, nó sẽ kiểm tra giá trị của GoogleAnalyticsObject, mà trước đó đã được thiết lập để trỏ đến tên của gavới dòng

 window['GoogleAnalyticsObject'] = 'ga';

Google đã xuất bản phiên bản chưa được rút gọn của mã này:

<!-- Google Analytics -->
<script>
/**
 * Creates a temporary global ga object and loads analytics.js.
 * Parameters o, a, and m are all used internally. They could have been
 * declared using 'var', instead they are declared as parameters to save
 * 4 bytes ('var ').
 *
 * @param {Window}        i The global context object.
 * @param {HTMLDocument}  s The DOM document object.
 * @param {string}        o Must be 'script'.
 * @param {string}        g Protocol relative URL of the analytics.js script.
 * @param {string}        r Global name of analytics object. Defaults to 'ga'.
 * @param {HTMLElement}   a Async script tag.
 * @param {HTMLElement}   m First script tag in document.
 */
(function(i, s, o, g, r, a, m){
  i['GoogleAnalyticsObject'] = r; // Acts as a pointer to support renaming.

  // Creates an initial ga() function.
  // The queued commands will be executed once analytics.js loads.
  i[r] = i[r] || function() {
    (i[r].q = i[r].q || []).push(arguments)
  },

  // Sets the time (as an integer) this tag was executed.
  // Used for timing hits.
  i[r].l = 1 * new Date();

  // Insert the script tag asynchronously.
  // Inserts above current tag to prevent blocking in addition to using the
  // async attribute.
  a = s.createElement(o),
  m = s.getElementsByTagName(o)[0];
  a.async = 1;
  a.src = g;
  m.parentNode.insertBefore(a, m)
})(window, document, 'script', '//www.google-analytics.com/analytics.js', 'ga');

// Creates a default tracker with automatic cookie domain configuration.
ga('create', 'UA-XXXXX-Y', 'auto');

// Sends a pageview hit from the tracker just created.
ga('send', 'pageview');
</script>
<!-- End Google Analytics -->

i['GoogleAnalyticsObject']=r; Điều này đang gán 'ga' cho thuộc tính 'GoogleAnalyticsObject' của 'cửa sổ'

window['ga'] = window['ga'] || function(){
        (window['ga'].q = window['ga'].q || []).push(arguments)
    }, window['ga'].l = 1 * new Date();

Phần này chỉ định thuộc tính 'ga' của cửa sổ làm hàm (hoặc chính nó nếu nó đã tồn tại). Hàm này sau đó gán thuộc tính q làm một mảng trống và thêm tất cả các đối số hàm vào nó. Sau đó nó gán thuộc tính l là dấu thời gian hiện tại (nó được nhân với 1 để ép buộc nó làm một số nguyên).

Các dòng tiếp theo chỉ cần tạo một thẻ script và gán cho nó một số thuộc tính như source và async = true và sau đó nó thêm thẻ script này vào tài liệu.
